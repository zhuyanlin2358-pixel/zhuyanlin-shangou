⚡ 这是动效组件专用模板，与静态组件模板的核心区别：动画循环管理 和 暂停后导出。 如果你做的是静态切图，用 静态模板 即可。

与静态组件模板的区别

动效专有
requestAnimationFrame 动画循环
需要维护一个持续运行的动画帧循环，并在需要时能暂停和恢复
动效专有
SG_EXPORT 必须先暂停再截图
收到导出请求时，先停止动画 → canvas.toDataURL() → 恢复动画。否则截到的是空帧
动效专有
SG_SET_CONFIG 实时更新动效参数
工具传来新配置（颜色/速度等），需要在不重启的情况下实时生效
与静态相同
Meta 标签、SG_READY、画布尺寸
sg-component / sg-name / sg-id / sg-canvas-width / sg-canvas-height 完全一样
注意
page 隐藏时自动暂停，恢复时继续
监听 visibilitychange，标签切走时停止动画避免性能浪费

完整模板代码

HTML · sg-animation-template.html 
<!DOCTYPE html>
<html lang="zh-CN">
<head>
<meta charset="UTF-8">

<!-- 必填 meta（与静态模板相同）-->
<meta name="sg-component"       content="true">
<meta name="sg-name"            content="动效名称">
<meta name="sg-id"              content="your-anim-id">
<meta name="sg-canvas-width"    content="750">
<meta name="sg-canvas-height"   content="424">
<meta name="sg-version"         content="1.0.0">
<meta name="sg-author"          content="你的名字">

<title>动效名称</title>
<style>
  * { box-sizing: border-box; margin: 0; padding: 0; }
  html, body { width: 100%; height: 100%; overflow: hidden; background: transparent; }
  #canvas { display: block; }
</style>
</head>
<body>

<canvas id="canvas" width="750" height="424"></canvas>

<script>
// ── 画布 ────────────────────────────────────────────────────────────────
const canvas = document.getElementById('canvas')
const ctx    = canvas.getContext('2d')

// ── 当前配置（工具通过 SG_SET_CONFIG 更新）──────────────────────────────
let config = {
  theme: {
    primaryColor: '#fad900',
    bgColor:      '#FFD9E5',
  },
  custom: {
    // ↓ 你的动效专属参数
    speed:    1.0,
    density:  50,
    color:    '#ff9ebc',
    duration: 3,
  },
}

// ── 动画状态 ─────────────────────────────────────────────────────────────
let animationId = null   // requestAnimationFrame 返回的 id
let isPaused    = false  // 是否已暂停

// ── 渲染单帧（你的核心动效逻辑写这里）──────────────────────────────────
function drawFrame(timestamp) {
  ctx.clearRect(0, 0, 750, 424)

  // ↓↓↓ 你的绘制代码 ↓↓↓
  ctx.fillStyle = config.theme.bgColor
  ctx.fillRect(0, 0, 750, 424)
  // ... 粒子、波纹、光晕等动效 ...
}

// ── 启动动画循环 ─────────────────────────────────────────────────────────
function startAnimation() {
  isPaused = false
  function loop(ts) {
    if (isPaused) return          // 暂停时直接退出循环
    drawFrame(ts)
    animationId = requestAnimationFrame(loop)
  }
  animationId = requestAnimationFrame(loop)
}

// ── 暂停动画循环 ─────────────────────────────────────────────────────────
function pauseAnimation() {
  isPaused = true
  if (animationId) {
    cancelAnimationFrame(animationId)
    animationId = null
  }
}

// ── 实时应用新配置（不重启动画）─────────────────────────────────────────
function applyConfig(newConfig) {
  // 深合并，不替换整个对象
  if (newConfig.theme)  config.theme  = { ...config.theme,  ...newConfig.theme }
  if (newConfig.custom) config.custom = { ...config.custom, ...newConfig.custom }
  // 动效参数立刻在下一帧生效，不需要重启
}

// ── 监听消息（工具 → 动效）──────────────────────────────────────────────
window.addEventListener('message', (e) => {
  const { type, payload } = e.data || {}

  switch (type) {

    // 工具传来新配置（颜色/速度/密度等）
    case 'SG_SET_CONFIG':
      applyConfig(payload)
      break

    // ⚠️ 关键：先暂停 → 截图 → 恢复，否则截到空帧
    case 'SG_EXPORT':
      pauseAnimation()
      drawFrame(0)                           // 确保当前帧已渲染
      const dataUrl = canvas.toDataURL('image/png')
      window.parent.postMessage({ type: 'SG_EXPORT_RESULT', dataUrl }, '*')
      startAnimation()                        // 恢复动画
      break

    // 工具请求预览截图（不暂停，直接截当前帧）
    case 'SG_PREVIEW':
      window.parent.postMessage({
        type: 'SG_PREVIEW_RESULT',
        dataUrl: canvas.toDataURL('image/png')
      }, '*')
      break
  }
})

// ── 页面隐藏时暂停，恢复时继续（节省性能）──────────────────────────────
document.addEventListener('visibilitychange', () => {
  if (document.hidden) pauseAnimation()
  else startAnimation()
})

// ── 启动并通知工具就绪 ───────────────────────────────────────────────────
window.addEventListener('load', () => {
  startAnimation()   // 先跑起来

  window.parent.postMessage({
    type: 'SG_READY',
    meta: {
      name:   document.querySelector('meta[name="sg-name"]').content,
      id:     document.querySelector('meta[name="sg-id"]').content,
      width:  750,
      height: 424,
    },
  }, '*')
})
</script>
</body>
</html>

三个关键点详解

① 动画循环：isPaused 标志位控制
不要用 clearInterval，用 cancelAnimationFrame + isPaused。 暂停时设 isPaused = true，loop 函数在下一帧检测到就退出，不留僵尸帧。
② SG_EXPORT：暂停 → drawFrame(0) → 截图 → 恢复
收到 SG_EXPORT 时动画可能刚好在两帧之间，canvas 内容不完整。 先 pauseAnimation() → 再 drawFrame(0) 确保完整帧 → 再截图，截完后 startAnimation() 恢复。
③ SG_SET_CONFIG：深合并，不重启
工具实时传颜色/主题变化时，只更新 config 对象，不要重启动画。 用 { ...config.custom, ...newConfig.custom } 合并， 下一个 drawFrame 自动读新值生效。
